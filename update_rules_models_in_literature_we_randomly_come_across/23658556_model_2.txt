SHR =  (  SHR OR NOT IAA5 ) AND NOT CYA
SCR =  (  SYS OR JKD ) AND NOT CYA
JKD = MGP AND  (  SYS OR NOT PHB ) OR SYS AND JKD AND NOT PHB
MGP = SYM OR JYA
miRNA165 = SYS OR miRNA165
PHB = NOT miRNA165 AND NOT CYA
Auxin = NOT SHR OR Auxin AND SCR
IAA5 = NOT  (  Auxin AND NOT WOX5 )
WOX5 = PYI5 AND NOT CYA AND JKD
CLE = CLE AND NOT IAA5
ACR = CLE AND Auxin
SYS = SHR AND SCR
CYA = CLE AND ACR
PYI5 = NOT PHB AND NOT IAA5
JYA = JKD AND NOT Auxin
SYM = SYS AND NOT MGP
